9void EQCurveView::draw(VSTGUI::CDrawContext* dc)
10{
11 dc->setDrawMode(VSTGUI::kAntiAliasing);
12
13 drawFreqMarkers(dc);
14 drawDbMarkers(dc);
15
16 dc->setLineWidth(2);
17 dc->setFrameColor(VSTGUI::CColor(0, 255, 0, 255));
18
19 const float centerFreq = expf(logf(EQ::minFreq) + _shift * (logf(EQ::maxFreq) - logf(EQ::minFreq)));
20 const float flip = 1.0f - 2.0f * _direction;
21 const float dBpeakGain = flip * _intensity * EQ::dBMaxBoost;
22
23 const auto path = dc->createGraphicsPath();
24
25 const int numPoints = static_cast<int>(getWidth());
26 for (int i = 0; i < numPoints; ++i)
27 {
28 const float normX = static_cast<float>(i) / numPoints;
29 const float logFreq = log10f(EQ::minFreq) + normX * (log10f(EQ::maxFreq) - log10f(EQ::minFreq));
30 const float freq = powf(10.0f, logFreq);
31
32 const float x = (freq - centerFreq) / (centerFreq * _width);
33 float shape = shapeFromPreset(_shape, x);
34 if (shape < 0.0f) shape = 0.0f;
35
36 const float dBgain = dBpeakGain * shape;
37
38 float px = freqToX(freq);
39 float py = dbToY(dBgain);
40
41 if (i == 0)
42 path->beginSubpath({px, py});
43 else
44 path->addLine({px, py});
45 }
46
47 dc->drawGraphicsPath(path, VSTGUI::CDrawContext::kPathStroked);
48 dc->drawGraphicsPath(path, VSTGUI::CDrawContext::kPathStroked);
49 path->forget();
50}

68void EQCurveView::onParamChanged(const Steinberg::Vst::ParamID id, const float normalizedValue)
69{
70 switch (id)
71 {
72 case kParamShift:
73 _shift = normalizedValue;
74 break;
75 case kParamWidth:
76 _width = normalizedValue;
77 break;
78 case kParamIntensity:
79 _intensity = normalizedValue;

81 case kParamDirection:
82 _direction = normalizedValue;
83 break;
84 case kParamShape:
85 {
86 int maxShapeIndex = static_cast<int>(EqShapePreset::Count) - 1;
87 int shapeIndex = static_cast<int>(normalizedValue * maxShapeIndex + 0.5f);

89 _shape = static_cast<EqShapePreset>(shapeIndex);

91 }

97float EQCurveView::freqToX(const float freq) const
98{
99 const float normX = (log10f(freq) - log10f(EQ::minFreq))/ (log10f(EQ::maxFreq) - log10f(EQ::minFreq));
100 return normX * getWidth();
101}

103float EQCurveView::dbToY(const float db) const
104{
105 constexpr float visualMaxDB = EQ::dBMaxBoost + 8.0f;
106 const float drawableHeight = getHeight() - 2 * _padding;
107
108 const float normY = (visualMaxDB - db) / (2 * visualMaxDB);
109 return _padding + normY * drawableHeight;
110}

112void EQCurveView::drawFreqMarkers(VSTGUI::CDrawContext* dc) const
113{
114 dc->setLineWidth(1);
115 dc->setFrameColor(VSTGUI::CColor(100, 100, 100, 128));
116 dc->setFont(VSTGUI::kNormalFontVerySmall);
117 dc->setFontColor(VSTGUI::CColor(180, 180, 180, 255));
118
119 static const std::vector<float> freqMarkers = {
120 30.0f, 50.0f, 100.0f, 200.0f, 500.0f,
121 1000.0f, 2000.0f, 5000.0f, 10000.0f
122 };
123
124 for (const float freq : freqMarkers)
125 {
126 if (freq < EQ::minFreq || freq > EQ::maxFreq)
127 continue;
128
129 float x = freqToX(freq);
130 dc->drawLine(VSTGUI::CPoint(x, 0), VSTGUI::CPoint(x, getHeight()));
131
132 std::string label;
133 if (freq >= 1000.0f)
134 label = std::to_string(static_cast<int>(freq / 1000.0f)) + "k";
135 else
136 label = std::to_string(static_cast<int>(freq));
137
138 VSTGUI::CRect textRect(x - 15, getHeight() - 15, x + 15, getHeight());
139 dc->drawString(label.c_str(), textRect, VSTGUI::kCenterText);
140 }
