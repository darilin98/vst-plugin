36inline float shapeFromPreset(const EqShapePreset preset, const float x) {
37 switch (preset)
38 {
39 case LeftRiser:
40 return 1.5f - powf(x, 2.0f);
41 case SoftNotch:
42 return (expf(5.0f * x) - 1.0f) * (expf(5.0f * (1.0f - x)) - 1.0f) / powf(expf(5.0f * 0.5f) - 1.0f, 2.0f);
43 case Wave:
44 return -powf(x, 4.0f) + powf(x, 3.0f) + powf(x, 2.0f) + 0.5f;
45 case Valley:
46 return -0.27f * powf(x, 4.0f) + 0.6f * powf(x, 3.0f) + 0.6f * powf(x, 2.0f) - x;
47 default:
48 return 0.0f;
49 }
