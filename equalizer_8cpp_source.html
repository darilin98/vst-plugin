6void PolynomialEqualizer::modulate(fftwf_complex *freq_bins, int fft_size, int sample_rate)
7{
8 constexpr float minFreq = EQ::minFreq;
9 const float maxFreq = sample_rate * 0.5f; // Nyquist frequency
10 constexpr float dBMaxBoost = EQ::dBMaxBoost;
11
12 const float centerFreq = expf( logf(minFreq) + param_shift_ * (logf(maxFreq) - logf(minFreq)) ); // Knob has a logarithmic feel -> converting back to linear
13 const float flip = 1.0f - 2.0f * param_direction_;
14 const float dBpeakGain = flip * param_intensity_ * dBMaxBoost;
15
16 for (int i = 0; i < fft_size / 2 + 1; ++i) {
17 float freq = (i * sample_rate) / fft_size;
18
19 // Normalize curve so that high/low frequencies get treated equally
20 float x = (freq - centerFreq) / (centerFreq * param_width_);
21
22 float shape = shapeFromPreset(param_shape_, x);
23 if (shape < 0.0f) shape = 0.0f;
24
25 float dBgain = dBpeakGain * shape;
26
27 float gain = powf(10.0f, dBgain * 0.05f);
28
29 freq_bins[i][0] *= gain; // real part
30 freq_bins[i][1] *= gain; // imag part
31 }
32}
