32void FFTProcessor::prepare(int32_t fft_size)
33{
34 fft_size_ = fft_size;
35 hop_size_ = fft_size_ / 2;
36 input_buffer_.assign(fft_size_, 0.0f);
37
38 in_ = (float*)fftwf_malloc(sizeof(float) * fft_size_);
39 out_ = (fftwf_complex*)fftwf_malloc(sizeof(fftwf_complex) * (fft_size_ /2 + 1));
40 processed_ = (float*)fftwf_malloc(sizeof(float) * fft_size_);
41
42 plan_fwd_ = fftwf_plan_dft_r2c_1d(fft_size_, in_, out_, FFTW_MEASURE);
43 plan_inv_ = fftwf_plan_dft_c2r_1d(fft_size_, out_, processed_, FFTW_MEASURE);
44
45 // Hann window
46 window_.resize(fft_size_);
47 for (int i = 0; i < fft_size_; ++i)
48 window_[i] = 0.5f * (1.0f - cosf((2.0f * M_PI * i) / (fft_size_ - 1.0f)));
49 overlap_add_buffer_.resize(fft_size_ + hop_size_, 0.0f);
50 write_pos_ = 0;
51}

61void FFTProcessor::process(float *input, float *output, float sample_rate, Steinberg::int32 num_samples)
62{
63 if (!in_ || !out_ || !processed_ || !plan_fwd_ || !plan_inv_)
64 return;
65
66 int samples_written = 0;
67
68 while (samples_written < num_samples) {
69 int new_samples = std::min<int>(num_samples - samples_written, hop_size_);
70
71 std::memmove(input_buffer_.data(),
72 input_buffer_.data() + hop_size_,
73 sizeof(float) * (fft_size_ - hop_size_));
74
75 std::copy(input + samples_written,
76 input + samples_written + new_samples,
77 input_buffer_.data() + (fft_size_ - hop_size_));
78
79 for (int i = 0; i < fft_size_; ++i)
80 in_[i] = input_buffer_[i] * window_[i];
81
82 fftwf_execute(plan_fwd_);
83
84 if (equalizer_) {
85 equalizer_->modulate(out_, fft_size_, static_cast<int>(sample_rate));
86 }
87
88 fftwf_execute(plan_inv_);
89
90 for (int i = 0; i < fft_size_; ++i) {
91 float sample = (processed_[i] / fft_size_);
92 overlap_add_buffer_[(write_pos_ + i) % overlap_add_buffer_.size()] += sample;
93 }
94
95 for (int i = 0; i < hop_size_; ++i) {
96 output[samples_written + i] =
97 overlap_add_buffer_[(write_pos_ + i) % overlap_add_buffer_.size()];
98 overlap_add_buffer_[(write_pos_ + i) % overlap_add_buffer_.size()] = 0.f;
99 }
100
101 write_pos_ = (write_pos_ + hop_size_) % overlap_add_buffer_.size();
102 samples_written += hop_size_;
103 }
104}
