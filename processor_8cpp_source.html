90bool PluginProcessor::getBypassState(const ProcessData &data) const
91{
92 if (!data.inputParameterChanges)
93 return bypassState == 1.0f;
94
95 for (int32 i = 0; i < data.inputParameterChanges->getParameterCount(); ++i)
96 {
97 auto* queue = data.inputParameterChanges->getParameterData(i);
98 if (!queue || queue->getParameterId() != kParamBypass)
99 continue;
100
101 int32 sampleOffset;
102 ParamValue value;
103
104 if(queue->getPoint(queue->getPointCount() - 1, sampleOffset, value) != kResultTrue)
105 continue;
106
107 bool newBypassState = value == 1.0f;
108
109 if (!(bypassState == 1.0f) && newBypassState) {
110 for (auto& fft : fft_processors_)
111 if (fft) fft->reset();
112 }
113
114 const_cast<PluginProcessor*>(this)->bypassState = value;
115 return newBypassState;
116 }
117
118 return bypassState == 1.0f;
119}
